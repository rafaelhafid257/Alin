import numpy as np

A = np.array([
    [1, -1, 1, 3],
    [2, 1, 3, 8],
    [6, 0, -1, 4]
], dtype=float)

n = A.shape[0]

for i in range(n):
    A[i] = A[i] / A[i][i]

    for j in range(i + 1, n):
        A[j] = A[j] - A[j][i] * A[i]

x = np.zeros(n)
for i in range(n - 1, -1, -1):
    x[i] = A[i, -1] - np.dot(A[i, i + 1:n], x[i + 1:])

print("Mengurangi matrix:")
print(A)
print("Solusi dari vector:")
print(x)
